#include <ESP8266WiFi.h>

const char* ssid = "TP-Link_6066"; // nombre red wifi
const char* pass = "72403358"; // contrase√±a de la red wifi

#define RED 4
#define GREEN 0
#define BLUE 2
WiFiServer server(80);

void setup() {
	pinMode(RED, OUTPUT);
	pinMode(GREEN, OUTPUT);
	pinMode(BLUE, OUTPUT);
	digitalWrite(RED, LOW);
	digitalWrite(GREEN, LOW);
	digitalWrite(BLUE, LOW);
	
	Serial.begin(115200);
	delay(10);

	/*Conexion WiFi*/
	Serial.println();
	Serial.println("Conectando a la red ");
	Serial.println(ssid);

	WiFi.begin(ssid,pass); // Inicia la conexion
	while (WiFi.status() != WL_CONNECTED){
		delay(500);
		Serial.println(".");
	}
	Serial.println();
	Serial.println("WiFi conectado");

	/*Inicia el server*/	
	sever.begin();
	Serial.println("inicia el servidor: ");
	Serial.print("Para conectarse, utilizar la siguiente URL: ");
	Serial.print("http://");
	Serial.print(WiFi.localIP());
	Serial.println("/");
}

// loop
void loop(){
	WiFiClient client = server.available();
	if(!client){
		return;
	}
	// Ya encontrado el cliente
	Serial.println("Nuevo cliente");
	while(!client.available()){ // Espera para abrir dato del cliente
		delay(1);
	}
	String request = client.readStringUntil('\r');
	Serial.println(request);
	client.flush();

	// Asocia request a la accion
	int value = LOW;
	if(request.indexOf("\LED=ON") != -1) {
		digitalWrite(RED, HIGH);
		value = HIGH;
	}
	if(request.indexOf("\LED=OFF") != -1) {
		digitalWrite(RED, LOW);
		value = LOW;
	}

	/*Retorna una respuesta: */
	client.println("http/1.1 200 OK");
	client.println("Content-Type: text/html");
	client.println("");
	client.println("<!DOCTYPE HTML>");
	client.println("<html>");

	client.print("El LED esta ");

	if (value == HIGH) {
		client.print("ON - Encendido");
	}
	else {
		client.print("OFF - Apagado");
	}
	client.println("<br><br>");
	client.println("<a href=\"/LED=ON\"><button> Encender </button></a>");
	client.println("<a href=\"/LED=OFF\"><button> Apagar </button></a>");
	client.println("</html>");

	delay(1);
	Serial.println("Cliente desconectado");
	Serial.println("");
}
